313 std::string generate_fixed_length_hash(size_t length) const {
314 static const char charset[] =
315 "0123456789"
316 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
317 "abcdefghijklmnopqrstuvwxyz";
318 static thread_local std::mt19937 generator(std::random_device{}());
319 static thread_local std::uniform_int_distribution<size_t> distribution(0, sizeof(charset) - 2);
320
321 std::string hash;
322 hash.reserve(length);
323 for (size_t i = 0; i < length; ++i) {
324 hash += charset[distribution(generator)];
325 }
326 return hash;
327 }

330 void remove_old_logs() {
331 const int64_t threshold_ts = time_shield::ms_to_sec(current_timestamp_ms()) - (m_config.auto_delete_days * time_shield::SEC_PER_DAY);
332# if __cplusplus >= 201703L
333 fs::path dir_path(get_directory_path());
334 if (!fs::exists(dir_path) || !fs::is_directory(dir_path)) {
335 return;
336 }
337 for (const auto& entry : fs::directory_iterator(dir_path)) {
338 if (!fs::is_regular_file(entry.status())) continue;
339 std::string filename = entry.path().filename().string();
340 if (is_valid_log_filename(filename)) {
341 const int64_t file_ts = get_timestamp_from_filename(filename);
342 if (file_ts < threshold_ts) {
343# if defined(_WIN32) || defined(_WIN64)
344 fs::remove(fs::path(utf8_to_ansi(entry.path().string())));
345# else
346 fs::remove(entry.path());
347# endif
348 }
349 }
350 }
351# else
352 const std::vector<std::string> file_list = get_list_files(get_directory_path());
353 for (const auto& file_path : file_list) {
354 std::string filename = get_file_name(file_path);
355 if (is_valid_log_filename(filename)) {
356 const int64_t file_ts = get_timestamp_from_filename(filename);
357 if (file_ts < threshold_ts) {
358# if defined(_WIN32) || defined(_WIN64)
359 remove(utf8_to_ansi(file_path).c_str());
360# else
