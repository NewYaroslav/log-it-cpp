356 std::string generate_fixed_length_hash(size_t length) const {
357 static const char charset[] =
358 "0123456789"
359 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
360 "abcdefghijklmnopqrstuvwxyz";
361 static thread_local std::mt19937 generator(std::random_device{}());
362 static thread_local std::uniform_int_distribution<size_t> distribution(0, sizeof(charset) - 2);
363
364 std::string hash;
365 hash.reserve(length);
366 for (size_t i = 0; i < length; ++i) {
367 hash += charset[distribution(generator)];
368 }
369 return hash;
370 }

373 void remove_old_logs() {
374 const int64_t threshold_ts =
375 time_shield::ms_to_sec(current_timestamp_ms()) -
376 (m_config.auto_delete_days * time_shield::SEC_PER_DAY);
377# if __cplusplus >= 201703L
378
379# if defined(_WIN32)
380 fs::path dir_path = fs::u8path(get_directory_path());
381# else
382 fs::path dir_path(get_directory_path());
383# endif
384
385 if (!fs::exists(dir_path) || !fs::is_directory(dir_path)) {
386 return;
387 }
388
389 for (const auto& entry : fs::directory_iterator(dir_path)) {
390 if (!fs::is_regular_file(entry.status())) continue;
391 std::string filename = entry.path().filename().string();
392 if (is_valid_log_filename(filename)) {
393 const int64_t file_ts = get_timestamp_from_filename(filename);
394 if (file_ts < threshold_ts) {
395 fs::remove(entry.path());
396 }
397 }
398 }
399# else
400 const std::vector<std::string> file_list = get_list_files(get_directory_path());
401 for (const auto& file_path : file_list) {
402 std::string filename = get_file_name(file_path);
403 if (is_valid_log_filename(filename)) {
404 const int64_t file_ts = get_timestamp_from_filename(filename);
405 if (file_ts < threshold_ts) {
406# if defined(_WIN32)
407 remove(utf8_to_ansi(file_path).c_str());
408# else
409 remove(file_path.c_str());
410# endif
411 }
412 }
413 }
414# endif
