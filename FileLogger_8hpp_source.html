155 void start_logging() {
156 // потоки ввода-вывода (например, std::cin, std::cout, std::cerr) могут быть закрыты до завершения программы.
157 // В этом случае вызовы функций, которые используют потоки ввода-вывода (например, конструктор std::regex),
158 // могут приводить к нежелательным поведениям, таким как зависание или ошибки сегментации.
159 is_valid_log_filename("2024-01-01.log");
160 std::lock_guard<std::mutex> lock(m_mutex);
161 try {
162 initialize_directory();
163 open_log_file(get_current_utc_date_ts());
164 remove_old_logs();
165 } catch (const std::exception& e) {
166 std::cerr << "Initialization error: " << e.what() << std::endl;
167 }
168 }

218 void write_log(const std::string& message, const int64_t& timestamp_ms) {
219 const int64_t message_date_ts = time_shield::start_of_day(time_shield::ms_to_sec(timestamp_ms));
220 if (message_date_ts != m_current_date_ts) {
221 open_log_file(message_date_ts);
222 remove_old_logs();
223 }
224 if (m_file.is_open()) {
225 m_file << message << std::endl;
226 }
227 }

230 void remove_old_logs() {
231 const int64_t threshold_ts = m_current_date_ts - (time_shield::SEC_PER_DAY * m_config.auto_delete_days);
232# if __cplusplus >= 201703L
233 fs::path dir_path(get_directory_path());
234 if (!fs::exists(dir_path) ||
235 !fs::is_directory(dir_path)) {
236 return;
237 }
238
239 for (const auto& entry : fs::directory_iterator(dir_path)) {
240 if (!fs::is_regular_file(entry.status())) continue;
241 std::string filename = entry.path().filename().string();
242 if (is_valid_log_filename(filename)) {
243 const int64_t file_date_ts = get_date_ts_from_filename(filename);
244 if (file_date_ts < threshold_ts) {
245 fs::remove(entry.path());
246 }
247 }
248 }
249# else
250 std::vector<std::string> file_list = get_list_files(get_directory_path());
251 for (const auto& file_path : file_list) {
252 // Извлекаем имя файла
253 std::string filename = file_path.substr(file_path.find_last_of("/\\") + 1);
254 if (is_valid_log_filename(filename)) {
255 const int64_t file_date_ts = get_date_ts_from_filename(filename);
256 if (file_date_ts < threshold_ts) {
257 remove(file_path.c_str());
258 }
259 }
260 }
261# endif
262 }
