156 void start_logging() {
157 // потоки ввода-вывода (например, std::cin, std::cout, std::cerr) могут быть закрыты до завершения программы.
158 // В этом случае вызовы функций, которые используют потоки ввода-вывода (например, конструктор std::regex),
159 // могут приводить к нежелательным поведениям, таким как зависание или ошибки сегментации.
160 is_valid_log_filename("2024-01-01.log");
161 std::lock_guard<std::mutex> lock(m_mutex);
162 try {
163 initialize_directory();
164 open_log_file(get_current_utc_date_ts());
165 remove_old_logs();
166 } catch (const std::exception& e) {
167 std::cerr << "Initialization error: " << e.what() << std::endl;
168 }
169 }

227 void write_log(const std::string& message, const int64_t& timestamp_ms) {
228 const int64_t message_date_ts = time_shield::start_of_day(time_shield::ms_to_sec(timestamp_ms));
229 if (message_date_ts != m_current_date_ts) {

232 if (m_file.is_open()) {

239 void remove_old_logs() {
240 const int64_t threshold_ts = m_current_date_ts - (time_shield::SEC_PER_DAY * m_config.auto_delete_days);
241# if __cplusplus >= 201703L
242 fs::path dir_path(get_directory_path());
243 if (!fs::exists(dir_path) ||
244 !fs::is_directory(dir_path)) {
245 return;
246 }
247
248 for (const auto& entry : fs::directory_iterator(dir_path)) {
249 if (!fs::is_regular_file(entry.status())) continue;
250 std::string filename = entry.path().filename().string();
251 if (is_valid_log_filename(filename)) {
252 const int64_t file_date_ts = get_date_ts_from_filename(filename);
253 if (file_date_ts < threshold_ts) {
254# if defined(_WIN32) || defined(_WIN64)
255 fs::remove(fs::path(utf8_to_ansi(entry.path().string())));
256# else
257 fs::remove(entry.path());
258# endif
259 }
260 }
261 }
262# else
263 std::vector<std::string> file_list = get_list_files(get_directory_path());
264 for (const auto& file_path : file_list) {
265 // Извлекаем имя файла
266 std::string filename = file_path.substr(file_path.find_last_of("/\\") + 1);
267 if (is_valid_log_filename(filename)) {
268 const int64_t file_date_ts = get_date_ts_from_filename(filename);
269 if (file_date_ts < threshold_ts) {
270# if defined(_WIN32) || defined(_WIN64)
271 remove(utf8_to_ansi(file_path).c_str());
272# else
273 remove(file_path.c_str());
274# endif
275 }
276 }
277 }
278# endif
