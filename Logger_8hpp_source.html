135 void log(const LogRecord& record) {
136 if (m_shutdown) return;
137 std::lock_guard<std::mutex> lock(m_mutex);
138 // Log to the specific logger if the index is valid
139 if (record.logger_index >= 0 && record.logger_index < static_cast<int>(m_loggers.size())) {
140 const auto& strategy = m_loggers[record.logger_index];
141 if (!strategy.enabled) return;
142 if (static_cast<int>(record.log_level) < static_cast<int>(strategy.logger->get_log_level())) return;
143 strategy.logger->log(record, strategy.formatter->format(record));
144 return;
145 }
146 for (const auto& strategy : m_loggers) {
147 if (strategy.single_mode) continue;
148 if (!strategy.enabled) continue;
149 if (static_cast<int>(record.log_level) < static_cast<int>(strategy.logger->get_log_level())) continue;
150 strategy.logger->log(record, strategy.formatter->format(record));
151 }
