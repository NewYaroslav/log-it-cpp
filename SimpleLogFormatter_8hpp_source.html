142 std::string escape_json_string(const std::string& input) const {
143 std::ostringstream oss;
144 for (char c : input) {
145 switch (c) {
146 case '"': oss << "\\\""; break;
147 case '\\': oss << "\\\\"; break;

149 case '\f': oss << "\\f"; break;
